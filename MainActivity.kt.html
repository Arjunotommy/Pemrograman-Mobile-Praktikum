package com.example.latihan

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.view.View
import android.widget.AdapterView
import android.widget.ListView
import android.widget.Toast
import kotlinx.android.synthetic.main.activity_main.*

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        var listview = findViewById<ListView>(R.id.listView)

        var list = mutableListOf<Model>()

        list.add(Model("Ruby",  "rubby description", "img7"))
        list.add(Model("Rails", "rails description", "img8"))
        list.add(Model("Python","python description","img10))
        list.add(Model("Java Sript","Java Script description","img11"))
        list.add(Model("PHP", "PHP description","img12"))

        listview.setOnItemClickListener { parent: AdapterView<*>, view: View, position: Int, id: Long ->
            if (position == 0){
                Toast.makeText( this@MainActivity, "you click in ruby!", Toast.LENGTH_LONG).show()
            }
            if (position == 1){
                Toast.makeText( this@MainActivity,"you click in rails!", Toast.LENGTH_LONG).show()
            }
            if (position == 2){
                Toast.makeText(  this@MainActivity, "you click in python!", Toast.LENGTH_LONG).show()
            }
            if (position == 3){
                Toast.makeText(  this@MainActivity, "you click in Java Script!", Toast.LENGTH_LONG).show()
            }
            if (position == 4){
                Toast.makeText( this@MainActivity, "you click in PHP!", Toast.LENGTH_LONG).show()
            }
        }
    }
}
